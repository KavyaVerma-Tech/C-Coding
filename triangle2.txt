#include <stdio.h>
void main()
{
    int a,b,c;
    printf("enter the side a:");
    scanf("%d",&a);
    printf("enter the side b:");
    scanf("%d",&b);
    printf("enter the side c:");
    scanf("%d",&c);
    if(a + b + c == 180)
    {
    if(a==b && b==c)
    {
        printf("equilateral triangle");
    }
    else if(a==b || b==c || c==a)
    {
        printf("isosceles triangle");
    }
    else
    {
        printf("scalene triangle");
    }
    }
    else
    {
        printf("not a valid triangle");
    }
    getch();
}
