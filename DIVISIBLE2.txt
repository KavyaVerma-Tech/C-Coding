#include <stdio.h>
void main()
{
    int num;
    printf("enter the number:");
    scanf("%d",&num);
    if(num % 3 == 0 && num % 7 == 0)
    {
        printf("number is divisible by 3 and 7");
    }
    else
    {
        printf("number is not divisible by 3 and 7");
    }
    getch();
}
