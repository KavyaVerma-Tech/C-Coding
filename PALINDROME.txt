#include <stdio.h>
void main()
{
    int num,orig_num,mod,rev=0;
    printf("enter the number:");
    scanf("%d",&num);
    orig_num = num;
    while(num>0)
    {
        mod = num % 10;
        num = num / 10;
        rev = rev * 10 + mod;
    }
    printf("reverse number = %d\n",rev);
    if(orig_num==rev)
    {
        printf("number is palindrome");
    }
    else
    {
        printf("number is not palindrome");
    }
    getch();
}
