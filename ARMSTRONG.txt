#include<stdio.h>
#include<math.h>
int main()
{
    int num, mod, digit, orig_num, n=0, result=0;
    printf("enter the number:");
    scanf("%d",&num);
    orig_num = num;
    while(num>0)
    {
        num = num / 10;
        ++n;
    }
    num = orig_num;
    while(num>0)
    {
        mod = num % 10;
        digit = pow(mod,n);
        result = result + digit;
        num = num / 10;
    }
    if(orig_num==result)
    {
        printf("number is armstrong");
    }
    else
    {
        printf("number is not armstrong");
    }
    getch();
}
