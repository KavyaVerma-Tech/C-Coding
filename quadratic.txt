#include <stdio.h>
#include <math.h> 
void main() 
{
    double a, b, c, D, r1, r2, real, imaginary;
    printf("Enter coefficients a, b, and c: ");
    scanf("%lf %lf %lf", &a, &b, &c);
    D = b * b - 4 * a * c;
    if(D>0)    
    {
        r1 = (-b + sqrt(D)) / (2 * a);
        r2 = (-b - sqrt(D)) / (2 * a);
        printf("Roots are real and distinct\n");
        printf("Roots are %.2lf and %.2lf",r1, r2);
    } 
    else if (D==0) 
    {
        r1 = r2 = -b / (2 * a);
        printf("Roots are real and equal\n");
        printf("Root 1 = Root 2 = %.2lf", r1);
    } 
    else 
    {
        real = -b / (2 * a);
        imaginary = sqrt(-D) / (2 * a);
        printf("roots are imaginary\n");
        printf("roots are %.2lf + %.2lfi",real, imaginary);
    }
    getch();
}
